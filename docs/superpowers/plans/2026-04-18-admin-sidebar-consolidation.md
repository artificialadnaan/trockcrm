- [ ] **Step 2: Add visibility and active-route helpers**

Add minimal helpers inside the same file:

```tsx
function filterByRole(items: NavItem[], role: Role | undefined) {
  if (!role) return [];
  return items.filter((item) => item.roles.includes(role));
}

function isGroupActive(items: NavItem[], pathname: string) {
  return items.some((item) => pathname === item.to || pathname.startsWith(`${item.to}/`));
}
```

Use `useLocation()` from `react-router-dom` so active-group logic is based on the current route.

## Task 3: Implement Collapsible Group Rendering and Active-State Rules

**Files:**

- [ ] **Step 1: Render collapsible admin groups with forced-open active behavior**

Inside `Sidebar`, add state and derived visibility:

```tsx
const { pathname } = useLocation();
const role = user?.role;

const visibleAdminGroups = adminGroups
  .map((group) => ({
    ...group,
    items: filterByRole(group.items, role),
  }))
  .filter((group) => group.items.length > 0);

const [expandedGroups, setExpandedGroups] = useState<Record<string, boolean>>({});

useEffect(() => {
  setExpandedGroups((current) => {
    const next = { ...current };
    for (const group of visibleAdminGroups) {
      if (isGroupActive(group.items, pathname)) {
        next[group.id] = true;
      } else if (!(group.id in next)) {
        next[group.id] = group.defaultExpanded;
      }
    }
    return next;
  });
}, [pathname, visibleAdminGroups]);

function isExpanded(group: AdminGroup & { items: NavItem[] }) {
  return isGroupActive(group.items, pathname) || expandedGroups[group.id] || false;
}
```

Render each group header as a button with `aria-expanded`, then render its visible items only when expanded.

- [ ] **Step 2: Prevent manual collapse of the active group while allowing other groups to toggle**

Use a toggle handler like:

```tsx
function toggleGroup(group: AdminGroup & { items: NavItem[] }) {
  if (isGroupActive(group.items, pathname)) return;
  setExpandedGroups((current) => ({
    ...current,
    [group.id]: !isExpanded(group),
  }));
}
```

This matches the spec’s route-state precedence rule.
